## 6. 手动创建 gulpfile.js

## 7. 手动编辑 gulpfile.js
```javascript
	var gulp = require('gulp'),
		jade = require('gulp-jade');
		
	gulp.task('jade',function(){
		return gulp.src('src/template/**/*.jade')
			.pipe(jade())
			.pipe(gulp.dest('builds/development'));
	});
```
